static fillPrevs(DNode list) {
	DNode trav = list;
	DNode trail;
	
	trav.prev = null;
	trav = trav.next;
	
	while (trav != null) {
		trav.prev = trail;	
		trail = trav;
		trav = trav.next;
	}		
}
